\mt1 金剛獅子之修心八頌
\r 嘎當大善知識朗日唐巴
\c 1
\q
\b
\v 1 吾祝诸有情，
\q1 超胜如意宝，
\q1 成办殊胜利，
\q1 恆时愿顾惜。
\q
\b
\v 2 随处人结交，
\q1 视己最卑俗，
\q1 真诚予他人，
\q1 最极之珍爱。
\q
\b
\v 3 观察诸行持，
\q1 心生烦恼时，
\q1 不利自他故，
\q1 强行愿遮除。
\q
\b
\v 4 性恶之眾生，
\q1 屈就猛烈苦，
\q1 见时如意宝，
\q1 难得愿爱重。
\q
\b
\v 5 于我因妒嫉，
\q1 他人责谤等，
\q1 愿取非理亏，
\q1 胜利供奉他。
\q
\b
\v 6 凡吾所利济，
\q1 寄望之对象，
\q1 纵非理损害，
\q1 愿视為善土。
\q
\b
\v 7 直接与间接，
\q1 利乐施眾生，
\q1 害损痛苦等，
\q1 秘密愿自取。
\q
\b
\v 8 波等不沾染，
\q1 八法分别垢，
\q1 心知法如幻，
\q1 不耽愿离繫。
